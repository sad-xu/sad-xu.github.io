# 包含min函数的栈

定义栈的数据结构，请在该类型中实现一个能够得到栈中所含最小元素的min函数（时间复杂度应为O（1））。

function push(node) {
  // write code here
}
function pop() {
  // write code here
}
function min() {
  // write code here
}

# 解

let stack = []
function push(node) {
  stack.push(node)
}
function pop() {
  return stack.pop()
}
function min() {
  if (stack.length === 0) return false
  let min = stack[0]
  stack.forEach(item => {
    min = Math.min(min, item)
  })
  return min
}
